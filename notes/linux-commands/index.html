Useful Linux Commands

Useful Linux Commands Link to heading

Process Management Link to heading 

# Find process by name
ps aux | grep process_name

# Kill process by PID
kill -9 PID

# Kill process by name
pkill process_name

File Operations Link to heading 

# Find files by name
find /path -name "filename"

# Find files by content
grep -r "search_term" /path

# Copy with progress
rsync -av --progress source/ destination/

System Information Link to heading 

# Check disk usage
df -h

# Check memory usage
free -h

# Check CPU info
lscpu

# Check system load
uptime

Network Link to heading 

# Check open ports
netstat -tulpn

# Check network interfaces
ip addr show

# Test connectivity
ping hostname

Package Management (Ubuntu/Debian) Link to heading 

# Update package list
sudo apt update

# Upgrade packages
sudo apt upgrade

# Install package
sudo apt install package_name

# Remove package
sudo apt remove package_name
